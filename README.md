## Usage

Simple Usage
```js
const { Client } = require('whatsapp-web.js');

client.on('message', msg => {
    if (msg.body == '!ping') {
        msg.reply('pong');
    }
    else if(msg.body){
        msg.reply('*BOT WA Auto Reply*\nPesan Anda akan dibalas secara berurutan dari bawah')
    }
});
```
